    {

                let countCharactersofD = 0;
                
                for (let i = 0; i < S.length; i++) {
                    if (S[i] === "A") {
                    countCharactersofA++;
                    } else if (S[i] === "D") {
                    countCharactersofD++;
                    }
                }
                
                return [countCharactersofA, countCharactersofD];
    };

Q2 - Count the Heads

        var Count_Occ = (s) => 
        {
              let CountHead = {};
            for (let i = 0; i < s.length; i++) {
                let findnochar = s.charAt(i);
                if (CountHead[findnochar]) {
                    CountHead[findnochar]++;
                } else {
                    CountHead[findnochar] = 1;
                }
            }
            let result = "";
            let K = Object.keys(CountHead).sort();
            for (let i = 0; i < K.length; i++) {
                if (CountHead[K[i]] > 1) {
                    result += K[i] + CountHead[K[i]];
                }
            }
        
            return result;
        };
