Q1 - Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
        {
              let hour = (h % 12) * 30 + m / 2;
              let minute = m * 6;
              let aDiff = Math.abs(hour - minute);
              let a = Math.min(aDiff, 360 - aDiff);
              return a;   
        };

Q3 - Perfect Number Check.

         
        var Perfect_Check = (N) => 
        {
              let Check = 0;
              let heard = 1;
              while(N>=heard) {
               if (N % heard === 0) {
                 Check += heard;
                }
                heard++;
              }
              if (Check === N) {
                return "YES";
              } else {
                return "NO";
              }    
        
        };

Q4 - Reverse a Number.

        var Reverse_Number = (N) => 
        {
        let reverse = 0;
        while (N > 0) {
            let R = N % 10;
            reverse = reverse * 10 + R;
            N = Math.floor(N / 10);
        }
        return reverse;

        };
